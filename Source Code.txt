import java.util.Scanner;
import java.util.Stack;

public class Main {
    public Main() {
    }

    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        new Stack();
        Stack<String> nameStack = new Stack();
        System.out.println("How many foods you want to store?");
        int array = input.nextInt();
        int[] arr = new int[array];

        int key;
        for(key = 0; key <= arr.length; ++key) {
            System.out.println("Enter your foods below: ");
            nameStack.add(input.nextLine());
        }

        System.out.println(nameStack);

        do {
            System.out.println("\n1. Add food \n2. Calculate \n3. Delete \n4. Sort \n5.Exit");
            key = input.nextInt();
            int count;
            if (key != 1) {
                int temp;
                if (key == 2) {
                    System.out.println("Choose how many calculation below: \n1. 2 \n2. 3 \n3. 4 \n4. 5");
                    count = input.nextInt();
                    int a;
                    int b;
                    if (count == 1) {
                        System.out.println("Enter your values");
                        a = input.nextInt();
                        b = input.nextInt();
                        temp = a + b;
                        System.out.println("the calories you have computed is done! \nThe sum of the calories is " + temp);
                    } else {
                        int c;
                        int d;
                        if (count == 3) {
                            System.out.println("Enter your values");
                            a = input.nextInt();
                            b = input.nextInt();
                            c = input.nextInt();
                            d = input.nextInt();
                            temp = a + b + c + d;
                            System.out.println("the calories you have computed is done! \nThe sum of the calories is " + temp);
                        } else if (count == 4) {
                            System.out.println("Enter your values");
                            a = input.nextInt();
                            b = input.nextInt();
                            c = input.nextInt();
                            d = input.nextInt();
                            int e = input.nextInt();
                            temp = a + b + c + d + e;
                            System.out.println("the calories you have computed is done! \nThe sum of the calories is " + temp);
                        }
                    }
                } else if (key == 3) {
                    System.out.println("Delete Successfully");
                    nameStack.pop();
                    System.out.println(nameStack);
                } else if (key == 4) {
                    Scanner scan = new Scanner(System.in);
                    System.out.print("Calorie to Organize ");
                    count = scan.nextInt();
                    int[] num = new int[count];
                    System.out.println("Enter Calorie: ");

                    int i;
                    for(i = 0; i < count; ++i) {
                        num[i] = scan.nextInt();
                    }

                    scan.close();

                    for(i = 0; i < count; ++i) {
                        for(int j = i + 1; j < count; ++j) {
                            if (num[i] > num[j]) {
                                temp = num[i];
                                num[i] = num[j];
                                num[j] = temp;
                            }
                        }
                    }

                    System.out.println("Calorie Ascending Order: ");

                    for(i = 0; i < count - 1; ++i) {
                        System.out.print(num[i] + ", ");
                    }

                    System.out.print(num[count - 1]);
                    return;
                }
        } while(key != 5);

    }
}
